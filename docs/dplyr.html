cars <- etl("mtcars")
cars %>%
  etl_create()
mtcars2 <- cars %>%
  tbl("mtcars")

Get some results

mtcars %>%
  group_by(cyl) %>%
  summarize(N = n(), avg_mpg = mean(mpg), 
            broken = paste(mean(mpg), mean(disp), sep = "-"))

  summarize(N = n(), avg_mpg = mean(mpg),
